# Constants

  • Access-anywhere
  • static (shared by all instances)

# Define - outside any class

define("PI",3.14);      // UPPERCASE
echo PI;                // 3.14
PI = 333;               // error [Read-only]

# Define - inside/outside any class

const PI = 3.14;
class A {  const PI = 3; }
$a = new A();
echo $a->PI;            // Gives nothing.
echo A::PI;             // 3
echo PI;                // 3.14
